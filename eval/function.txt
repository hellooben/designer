        return x + y!

    }!

    method biggest(a, b) {
        if (a > b) {
            return a!
        }
        else {
            return b!
        }!
    }!

        return add(1000, num)!
    }!

    var v = 20!
    var v1 = 10!
    var sum = add(biggest(v, v1), 50)!
    print(sum)!
    print()!

    var big = thousand(234)!
